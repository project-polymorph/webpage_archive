|/www/html
|___ /www/html/matomo

   server {
          listen 443;
          server_name www.example.com;
           root   /www/html;
            auth_basic "Please input password";
            auth_basic_user_file /usr/local/nginx/db/passwd.db;
            index  index.html index.htm index.php;

            location /matomo {
            #关闭顶级目录认证
            auth_basic off;          
            location = /matomo/piwik.js {
            #关闭piwik.js文件认证
            auth_basic off;
